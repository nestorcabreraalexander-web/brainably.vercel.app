    <script>

        // Interactive Timer Logic

        let timer, isRunning = false, timerType = 'study', timeRemaining = 1500;

        function updateTimerDisplay() {

            const min = Math.floor(timeRemaining / 60), sec = timeRemaining % 60;

            document.getElementById('timerDisplay').innerText = `${min.toString().padStart(2,'0')}:${sec.toString().padStart(2,'0')}`;

        }

        function toggleTimer() {

            const btn = document.getElementById('startBtn');

            if (isRunning) { clearInterval(timer); btn.innerText = 'Start'; isRunning = false; }

            else {

                isRunning = true; btn.innerText = 'Pause';

                timer = setInterval(() => { if (timeRemaining > 0) { timeRemaining--; updateTimerDisplay(); } else { clearInterval(timer); resetTimer(); } }, 1000);

            }

        }

        function resetTimer() { clearInterval(timer); isRunning = false; document.getElementById('startBtn').innerText = 'Start'; timeRemaining = timerType === 'study' ? 1500 : 300; updateTimerDisplay(); }

        function setTimerType(type) { timerType = type; document.getElementById('btn-study').classList.toggle('active', type==='study'); document.getElementById('btn-break').classList.toggle('active', type==='break'); resetTimer(); }

        function exportToTxt() { const txt = document.getElementById('chatDisplay').innerText; const blob = new Blob([txt], { type: "text/plain" }); const a = document.createElement("a"); a.download = "Brainably-Study-Guide.txt"; a.href = window.URL.createObjectURL(blob); document.body.appendChild(a); a.click(); document.body.removeChild(a); }
